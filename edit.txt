import { createTheme } from '@mui/material/styles';

const theme = createTheme({
  typography: {
    fontFamily: "IranianSans, IranianSansBold", // دقت کنید که بین نام فونت‌ها کاما قرار دهید
  },
  components: {
    // استایل‌دهی دکمه
    MuiButton: {
      styleOverrides: {
        root: ({ theme }) => ({ // استایل برای دکمه‌های معمولی
          borderRadius: 8, // شعاع گوشه‌های دکمه
          textTransform: 'none', // جلوگیری از تبدیل متن به حروف بزرگ
          fontWeight: 600,
          padding: '8px 16px', // padding دکمه
          // استایل‌های دیگر...
          '&:hover': { // استایل برای حالت hover
            backgroundColor: theme.palette.primary.dark, // تغییر رنگ پس‌زمینه در hover
          },
        }),
        contained: ({ theme }) => ({ // استایل برای دکمه‌های contained
          backgroundColor: theme.palette.primary.main,
          color: theme.palette.common.white,
          '&:hover': {
            backgroundColor: theme.palette.primary.dark,
          },
        }),
        outlined: ({ theme }) => ({ // استایل برای دکمه‌های outlined
          borderColor: theme.palette.primary.main,
          color: theme.palette.primary.main,
          '&:hover': {
            backgroundColor: alpha(theme.palette.primary.main, 0.08), // شفافیت در hover
          },
        }),
      },
    },
    // استایل‌دهی لینک
    MuiLink: {
      styleOverrides: {
        root: {
          textDecoration: 'none', // حذف underline پیش‌فرض لینک
          color: '#1976d2', // رنگ لینک
          '&:hover': {
            textDecoration: 'underline', // نمایش underline در hover
          },
        },
      },
    },
  },
});

export default theme;
